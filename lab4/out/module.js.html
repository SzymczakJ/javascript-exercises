class Operation {
  /**
   * 
   * @param {*} x x coord of operation
   * @param {*} y y coord of operation
   */
  constructor(x, y) {
    this.x = x;
    this.y = y;
  }
  /**
   * 
   * @returns sum of x and y
   */
  sum() {
    return this.x + this.y;
  }
}
module.exports = Operation;
